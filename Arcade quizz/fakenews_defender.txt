// ── PLANE ────────────────────────────────────────────────────
const plane = { x: W/2, y: H-55, speed:5 };

function drawPlane() {
  const x=plane.x, y=plane.y;
  ctx.shadowColor='#00ffcc'; ctx.shadowBlur=18;
  // Body
  ctx.fillStyle='#00ffcc';
  ctx.beginPath(); ctx.moveTo(x,y-18); ctx.lineTo(x+10,y+10); ctx.lineTo(x,y+4); ctx.lineTo(x-10,y+10); ctx.closePath(); ctx.fill();
  // Wings
  ctx.fillStyle='#0099aa';
  ctx.beginPath(); ctx.moveTo(x-28,y+12); ctx.lineTo(x-8,y+2); ctx.lineTo(x,y+10); ctx.lineTo(x-14,y+16); ctx.closePath(); ctx.fill();
  ctx.beginPath(); ctx.moveTo(x+28,y+12); ctx.lineTo(x+8,y+2); ctx.lineTo(x,y+10); ctx.lineTo(x+14,y+16); ctx.closePath(); ctx.fill();
  // Cockpit

// ── ANSWER BLOCKS ────────────────────────────────────────────
const COLORS = ['#ff6644','#aa44ff','#ffaa00','#00aaff'];

function shuffleColors() {
  const c = [...COLORS];
  for (let i = c.length-1; i > 0; i--) {
    const j = Math.floor(Math.random()*(i+1));
    [c[i],c[j]] = [c[j],c[i]];
  }
  return c;
}

function spawnAnswers() {
  answers = [];
  const q = QCM[qIndex];
  const cols = q.options.length;
  const blockW = 140, spacing = (W - cols*blockW) / (cols+1);
  const colors = shuffleColors();

  q.options.forEach((text, i) => {
    const x = spacing + i*(blockW+spacing) + blockW/2;
    answers.push({
      text,
      x,
      y: -60 - i*80,           // staggered start
      w: blockW, h: 46,
      isCorrect: i === q.answer,
      speed: 0.55 + Math.random()*0.2,   // base speed, multiplied live by speedMult
      color: colors[i],
      wobble: Math.random()*Math.PI*2,
      dead: false,
      flash: 0
    });
  });
}

function drawAnswer(a) {
  if (a.dead) return;
  const x=a.x, y=a.y, w=a.w, h=a.h;

  ctx.shadowColor = a.color;
  ctx.shadowBlur = a.flash > 0 ? 30 : 12;

  // Box
  ctx.fillStyle = a.flash > 0
    ? (a.isCorrect ? a.color : '#ff2222')
    : `${a.color}33`;
  ctx.strokeStyle = a.color;
  ctx.lineWidth = 1.8;
  roundRect(ctx, x-w/2, y-h/2, w, h, 10);
  ctx.fill(); ctx.stroke();

  // Text
  ctx.shadowBlur = 0;
  ctx.fillStyle = '#ffffff';
  ctx.font = `700 ${a.text.length > 22 ? '11' : '12.5'}px Nunito, sans-serif`;
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  wrapText(ctx, a.text, x, y, w-14, 15);
}

function wrapText(ctx, text, cx, cy, maxW, lineH) {
  const words = text.split(' ');
  let line='', lines=[];
  for (const w of words) {
    const test = line ? line+' '+w : w;
    if (ctx.measureText(test).width > maxW && line) { lines.push(line); line=w; }
    else line=test;
  }
  lines.push(line);
  const startY = cy - (lines.length-1)*lineH/2;
  lines.forEach((l,i) => ctx.fillText(l, cx, startY+i*lineH));
}

function roundRect(ctx, x,y,w,h,r) {
  ctx.beginPath();
  ctx.moveTo(x+r,y); ctx.lineTo(x+w-r,y);
  ctx.arcTo(x+w,y,x+w,y+r,r); ctx.lineTo(x+w,y+h-r);
  ctx.arcTo(x+w,y+h,x+w-r,y+h,r); ctx.lineTo(x+r,y+h);
  ctx.arcTo(x,y+h,x,y+h-r,r); ctx.lineTo(x,y+r);
  ctx.arcTo(x,y,x+r,y,r); ctx.closePath();
}

// ── PARTICLES ────────────────────────────────────────────────
function spawnParticles(x,y,color,count=14) {
  for (let i=0;i<count;i++) {
    const a=Math.random()*Math.PI*2, s=Math.random()*4+1;
    particles.push({ x,y, vx:Math.cos(a)*s, vy:Math.sin(a)*s,
      life:1, decay:Math.random()*.04+.02, r:Math.random()*4+2, color });
  }
}

// ── SOUND ────────────────────────────────────────────────────
function playSound(freq,dur,type='sine',vol=0.15) {
  try {
    const ac=new(window.AudioContext||window.webkitAudioContext)();
    const o=ac.createOscillator(), g=ac.createGain();
    o.type=type; o.frequency.setValueAtTime(freq,ac.currentTime);
    o.frequency.exponentialRampToValueAtTime(freq*.3,ac.currentTime+dur);
    g.gain.setValueAtTime(vol,ac.currentTime);
    g.gain.exponentialRampToValueAtTime(.001,ac.currentTime+dur);
    o.connect(g); g.connect(ac.destination); o.start(); o.stop(ac.currentTime+dur);
  } catch(e){}
}

// ── FEEDBACK ─────────────────────────────────────────────────
function showFeedback(msg, type) {
  const el=document.getElementById('feedback');
  el.textContent=msg; el.className='show '+type;
  clearTimeout(feedbackTimer);
  feedbackTimer=setTimeout(()=>{ el.className=''; }, 2800);
}

// ── QUESTION SETUP ───────────────────────────────────────────
function loadQuestion() {
  if (qIndex >= QCM.length) { endGame(true); return; }
  currentQ = QCM[qIndex];
  document.getElementById('question-text').textContent = currentQ.q;
  document.getElementById('q-num').textContent = qIndex+1;
  document.getElementById('question-banner').style.display='flex';
  spawnAnswers();
  questionActive = true;
}

function nextQuestion() {
  questionActive = false;
  qIndex++;
  setTimeout(loadQuestion, 1200);
}

// ── SHOOT ────────────────────────────────────────────────────
function shoot() {
  if (!gameRunning || !questionActive || shootCooldown>0) return;
  bullets.push({ x:plane.x, y:plane.y-20, speed:13 });
  shootCooldown=14;
  playSound(880,.05,'square');
}

// ── STARS ────────────────────────────────────────────────────
function makeStars() {
  stars=[];
  for (let i=0;i<100;i++)
    stars.push({ x:Math.random()*W, y:Math.random()*H,
      r:Math.random()*1.4+.3, speed:Math.random()*.4+.1, b:Math.random() });
}

// ── UPDATE UI ─────────────────────────────────────────────────
function updateUI() {
  document.getElementById('score-display').textContent=score;
  const h=['💀','❤️','❤️❤️','❤️❤️❤️'];
  document.getElementById('lives-display').textContent=h[Math.max(0,lives)]||'💀';
}

// ── START ────────────────────────────────────────────────────
function startGame() {
  document.getElementById('overlay').style.display='none';
  score=0; lives=3; qIndex=0; bullets=[]; answers=[]; particles=[];
  plane.x=W/2; shootCooldown=0; feedbackTimer=null;
  makeStars(); updateUI();
  gameRunning=true;
  loadQuestion();
  if(animId) cancelAnimationFrame(animId);
  loop();
}

// ── END ──────────────────────────────────────────────────────
function endGame(win) {
  gameRunning=false;
  document.getElementById('question-banner').style.display='none';
  const ov=document.getElementById('overlay');
  ov.style.display='flex';
  ov.innerHTML=`
    <h1 style="color:${win?'#00ffcc':'#ff4466'}">${win?'BRAVO !<br>MISSION ACCOMPLIE !':'GAME OVER'}</h1>
    <div class="sub">Score final : <strong style="color:#00ffcc">${score}</strong><br>
    ${win?'Tu es un vrai chasseur de Fake News !':'Continue à développer ton esprit critique !'}
    </div>
    <button onclick="startGame()">REJOUER ▶</button>
  `;
}

// ── MAIN LOOP ────────────────────────────────────────────────
function loop() {
  animId=requestAnimationFrame(loop);
  ctx.clearRect(0,0,W,H);

  // Stars
  stars.forEach(s=>{
    s.y+=s.speed; if(s.y>H){s.y=0;s.x=Math.random()*W;}
    ctx.globalAlpha=.3+s.b*.7;
    ctx.fillStyle='#fff'; ctx.beginPath(); ctx.arc(s.x,s.y,s.r,0,Math.PI*2); ctx.fill();
  });
  ctx.globalAlpha=1;

  // Scan lines
  ctx.strokeStyle='rgba(0,255,204,0.025)'; ctx.lineWidth=1;
  for(let y=0;y<H;y+=20){ ctx.beginPath();ctx.moveTo(0,y);ctx.lineTo(W,y);ctx.stroke(); }

  if(!gameRunning) return;

  // Input
  if(keys['ArrowLeft']||keys['KeyA']) plane.x=Math.max(34,plane.x-plane.speed);
  if(keys['ArrowRight']||keys['KeyD']) plane.x=Math.min(W-34,plane.x+plane.speed);
  if(keys['Space']) shoot();
  if(shootCooldown>0) shootCooldown--;

  // Bullets
  bullets = bullets.filter(b=>{
    b.y-=b.speed;
    if(b.y<-20) return false;
    // Draw
    ctx.shadowColor='#00ffff'; ctx.shadowBlur=12;
    const g=ctx.createLinearGradient(b.x,b.y,b.x,b.y+14);
    g.addColorStop(0,'#fff'); g.addColorStop(1,'#00ffcc');
    ctx.fillStyle=g;
    ctx.beginPath(); ctx.roundRect(b.x-2,b.y-7,4,14,2); ctx.fill();
    ctx.shadowBlur=0;

    // Check collision with answers
    let hit=false;
    if(questionActive) {
      answers.forEach(a=>{
        if(a.dead) return;
        if(!hit && Math.abs(b.x-a.x)<a.w/2+2 && Math.abs(b.y-a.y)<a.h/2+4) {
          hit=true;
          a.flash=15;
          spawnParticles(a.x,a.y,a.isCorrect?'#00ffcc':'#ff4466',18);

          if(a.isCorrect) {
            // Good shot!
            score+=50;
            updateUI();
            playSound(660,.15,'sine');
            showFeedback(currentQ.expl,'ok');
            setTimeout(()=>{ a.dead=true; nextQuestion(); }, 400);
          } else {
            // Wrong target!
            lives--;
            updateUI();
            playSound(150,.3,'sawtooth');
            showFeedback('❌ Mauvaise réponse ! Ne tire pas sur les leurres !','ko');
            if(lives<=0){ setTimeout(()=>endGame(false),600); return; }
            // Destroy wrong answer (penalised)
            setTimeout(()=>{ a.dead=true; },300);
          }
        }
      });
    }
    return !hit;
  });

  // Answers fall
  if(questionActive) {
    let allGone=true;
    answers.forEach(a=>{
      if(a.dead) return;
      allGone=false;
      a.wobble+=.012;
      a.x+=Math.sin(a.wobble)*0.5;
      a.y+=a.speed * speedMult;  // live speed — reacts instantly to slider
      if(a.flash>0) a.flash--;

      // Correct answer reaches bottom = lose a life
      if(a.y>H+30) {
        a.dead=true;
        if(a.isCorrect) {
          lives--;
          updateUI();
          playSound(150,.4,'sawtooth');
          showFeedback('💨 La bonne réponse s\'est échappée ! -1 vie','ko');
          if(lives<=0){ setTimeout(()=>endGame(false),600); return; }
          nextQuestion();
        }
      }
      drawAnswer(a);
    });
  }

  // Particles
  particles=particles.filter(p=>{
    p.x+=p.vx; p.y+=p.vy; p.vy+=.1; p.life-=p.decay;
    if(p.life<=0) return false;
    ctx.globalAlpha=p.life;
    ctx.fillStyle=p.color;
    ctx.beginPath(); ctx.arc(p.x,p.y,p.r*p.life,0,Math.PI*2); ctx.fill();
    ctx.globalAlpha=1;
    return true;
  });

  // Plane
  drawPlane();

  // Question number overlay bottom-right
  ctx.font='600 10px Orbitron';
  ctx.fillStyle='rgba(0,255,204,0.3)';
  ctx.textAlign='right'; ctx.textBaseline='bottom';
  ctx.fillText(`Q${qIndex+1}/10 — ${score}pts`, W-10, H-8);
}
